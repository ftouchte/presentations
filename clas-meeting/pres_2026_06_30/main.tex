    \begin{minipage}{0.55\linewidth}
        \begin{itemize}
            \item Principle: 
                \begin{itemize}
                    \item Select elastic events : electron + track (collection of hits)
                    \item Compute the expected track kinematics from the electron ones
                    \item Propagate the expected track
                    \item Estimate how far it deviates from the measured collection of hits
                \end{itemize}
            \item Our observable : {\color{red}residual\_LR}
                \begin{itemize}
                    \item It indicates whether the track is on right or on the left of the measurement
                \end{itemize}
            \item Correction angles (rotZ)
                \[
                    \alpha_i = \alpha_{i-1} - \frac{r_i}{R}\cdot \frac{1}{2}
                \]
        \end{itemize}
    \end{minipage}

    \begin{itemize}
        \item Limitations
            \begin{itemize}
                \item {\color{red}(CLAS electron vz)} has a small resolution for elastic events, $\theta_{e} \sim 6^\circ$
                \item {\color{blue}(vz shift)} relative position of the AHDC relative to the center of CLAS
            \end{itemize}

    \begin{itemize}
        \item Solutions/Observations
            \begin{itemize}
                \item Doing a fit can release the constraint on {\color{red}(CLAS electron vz)}
                \item Using the exact {\color{blue}(vz shift)} value make the correction angles smaller (so more physical)
            \end{itemize}
    \end{itemize}
