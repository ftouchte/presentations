\begin{frame}{Reminder - AHDC geometry}
    \begin{itemize}
        \item Cylindrical wire chamber filled with He/CO$_2$ (80/20\%)
        \item Novelty : 
            \begin{itemize}
                \item All wires have the same $\diameter$, 30 µm, and AlMg5 material
                \item Total of 3026 wires, 576 are readouts with positive charge
                %\item No guard wires (sense wire $\leftrightarrow$ HV, field wire $\leftrightarrow$ ground)
                \item Sense wires are distributed over 5 superlayers containing 1 or 2 layers (11, 21, 22, 31, 32, 41, 42, 51)
                \item It has stereo angles varying from 3.8$^\circ$ to 7.4$^\circ$ between superlayers for the vz reconstruction
            \end{itemize}
    \end{itemize}

\begin{frame}{AHDC alignment}
    \begin{itemize}
        \item Context: limitations in the AHDC track reconstruction
            \begin{itemize}
                \item need to improve the overall tracking resolution
                \item need to improve the resolution of the reconstructed $\theta$
                \item need to extra the relative position of the AHDC with respect to CLAS
            \end{itemize}
    \end{itemize}

    \begin{minipage}{0.55\linewidth}
        \begin{itemize}
            \item Principle: 
                \begin{itemize}
                    \item Select elastic events : electron + track (collection of hits)
                    \item Compute the expected track kinematics from the electron ones
                    \item Propagate the expected track
                    \item Estimate how far it deviates from the measured collection of hits
                \end{itemize}
            \item Our observable : {\color{red}residual\_LR}
                \begin{itemize}
                    \item It indicates whether the track is on right or on the left of the measurement
                \end{itemize}
            \item Correction angles (rotZ)
                \[
                    \alpha_i = \alpha_{i-1} - \frac{r_i}{R}\cdot \frac{1}{2}
                \]
        \end{itemize}
    \end{minipage}
    \begin{minipage}{0.42\linewidth}
        \begin{figure}
            \includegraphics[width=\linewidth]{figures/build/residual_LR_concept.pdf}
        \end{figure}
    \end{minipage}
    
\end{frame}
