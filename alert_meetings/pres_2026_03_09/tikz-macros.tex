\newcommand{\DrawCircleArcThroughThreePoints}[4][]{%
\path let
  \p1 = (#2),
  \p2 = (#3),
  \p3 = (#4)
in
\pgfextra{

\pgfmathsetmacro{\xA}{\x1}
\pgfmathsetmacro{\yA}{\y1}
\pgfmathsetmacro{\xB}{\x2}
\pgfmathsetmacro{\yB}{\y2}
\pgfmathsetmacro{\xC}{\x3}
\pgfmathsetmacro{\yC}{\y3}

% déterminant
\pgfmathsetmacro{\D}{2*(\xA*(\yB-\yC)+\xB*(\yC-\yA)+\xC*(\yA-\yB))}

% centre
\pgfmathsetmacro{\xc}{
((\xA*\xA+\yA*\yA)*(\yB-\yC) +
 (\xB*\xB+\yB*\yB)*(\yC-\yA) +
 (\xC*\xC+\yC*\yC)*(\yA-\yB))/\D
}

\pgfmathsetmacro{\yc}{
((\xA*\xA+\yA*\yA)*(\xC-\xB) +
 (\xB*\xB+\yB*\yB)*(\xA-\xC) +
 (\xC*\xC+\yC*\yC)*(\xB-\xA))/\D
}

% rayon
\pgfmathsetmacro{\r}{sqrt((\xA-\xc)^2+(\yA-\yc)^2)}

% angles
\pgfmathsetmacro{\angleA}{atan2(\yA-\yc,\xA-\xc)}
\pgfmathsetmacro{\angleC}{atan2(\yC-\yc,\xC-\xc)}

\draw[#1] (\xc,\yc) ++(\angleA:\r) arc(\angleA:\angleC:\r);

};
}
