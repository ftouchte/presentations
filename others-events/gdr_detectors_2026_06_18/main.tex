\begin{frame}{The ALERT detector}
    \begin{itemize}
        \item A requirement for a new physics program at Jefferson Lab
            \begin{itemize}
                \item physics process of interest: $eN \rightarrow eX\gamma$
                \item need to detect recoil fragments X : p, d, H$^3$, He$^3$, He$^4$
                \item kinematics range of interest:  $70 < p < 400$ MeV/c, $\theta > 45^\circ$
                \item the default CLAS12 setup can only detect particles with $p > 250$ MeV/c \footnote{$p = 250 $  MeV/c $\leftrightarrow$ E$_c \sim 31.25$ MeV for a proton}
            \end{itemize}
        \item We needed \textit{\color{red}A} (new) \textit{\color{red}L}ow \textit{\color{red}E}nergy \textit{\color{red}R}ecoil \textit{\color{red}T}racker
        \item The detector had to be placed at the center of the 5T solenoid
        \item Drift Chamber (IJCLab) + Time-Of-Flight (ANL)
    \end{itemize}

\begin{frame}{ALERT Hyperbolic Drift Chamber (AHDC)}
    \begin{itemize}
        \item Cylindrical wire chamber filled with He/CO$_2$ (80/20)
        \item Novelty : 
            \begin{itemize}
                \item All wires have the same $\diameter$, 30 µm, and AlMg5 material
                \item Total of 3026 wires, 576 are readouts with positive charge
                %\item No guard wires (sense wire $\leftrightarrow$ HV, field wire $\leftrightarrow$ ground)
                \item Superlayer structure : 3-5-5-5-3, resulting in 8 detection layers
                \item It has stereo angles varying from 3.8$^\circ$ to 7.4$^\circ$ between superlayers for the vz reconstruction
            \end{itemize}
        \item Use the DREAM electronics developed at CEA
    \end{itemize}

\begin{frame}{Track Finder}
    \begin{itemize}
        \item Clustering problem: find which collection of hits belong to a track
        \item Machine Learning approaches \footnote{Work by Mathieu Ouillon, see \href{https://indico.in2p3.fr/event/36031/contributions/157186/attachments/93301/142810/AG_GDR_061825.pdf}{GDR DI2I 2025}} based on MultiLayer Perceptron and Graph Neural Network are currently used
        \item The MLP infrastructure
            \begin{itemize}
                \item Inputs : X,Y coordinates of a set of hits
                \item Output : 0/1 (bad/good) track
            \end{itemize}
        \item The training is done on simulation
        \item Performance on real data: 96\% efficiency and 90\% purity for a 0.2 threshold
        %\item 
    \end{itemize}

\begin{frame}{Kalman Filter}
    \begin{itemize}
        \item We use the Kalman Filter to reconstruct the initial position and the momentum of the particle $\bm{x} = (x, y, z, p_x, p_y, p_z)^T$
            \begin{itemize}
                \item Prediction :  propagate the track towards the next AHDC hit
                \item Correction :  compute the residual and correct the estimation vector $\bm{\hat{x}}$
            \end{itemize}
        \item We already have good results whereas fine-tunings remain to be done:
            \begin{itemize}
                \item Good initialization of the error covariance matrix
                \item Precise process and measurement noises
                \item Number of iterations and computing time
            \end{itemize}
    \end{itemize}

\begin{frame}{AHDC resolution}
    \begin{itemize}
        \item The distribution of residuals \footnote{The fit is made with a Crystal Ball distribution, the width is the $\sigma$ parameter of the Gaussian part} gives an overall indication of the quality of our tracking reconstruction
        \item The integrated resolution is about 336 µm
        \item However, some layer have a better resolution! E.g. layer 42 has 261 µm
        \item This difference between layer is symptomatic of an unfinished calibration
    \end{itemize}
